这里用 C 实现一个 IP检查的方法。并对它进行测试验证。    


```
#include<stdio.h>

#define ANSI_COLOR_RED     "\x1b[31m"
#define ANSI_COLOR_RESET   "\x1b[0m"


//32位的ip地址合法性检查
int check_ip(char *ip){
    int a = 0, n=0;
    char c;

    while(  (c=*ip++) != '\0' ) {
        if(c=='.'){
            //检查a是否合法
            if(a>255 || a<0){
                return 0;
            }
            a = 0;
            n++;
        }else{
            int m = (c-'0');
            if(m>9 || m <0)
                return 0;
            a = a*10 + m;
        }
    } 
    //检查位数,n must be 4
    if(n!=3)
        return 0;
    return 1;
}

//定义测试用例
typedef struct testcase {
    int result;
    int expect;
    char *ip;
}testcase;

//验证单个测试用例
void valid_case(testcase a)
{
    a.result = check_ip(a.ip);
    if(a.expect != a.result)
        printf("Test case %s Failed %s, expect is %d, result is %d\n",ANSI_COLOR_RED, ANSI_COLOR_RESET, a.expect, a.result);
    else
        printf("Success for case %s\n", a.ip);
}

//测试用例执行
int test_check_ip()
{
    //valid ip
    testcase a1;
    a1.ip = "127.0.0.1";
    a1.expect = 1;
    valid_case(a1); 


    //invalid ip
    testcase a2 = { 1, 0 , "256.0.0.0"};
    valid_case(a2);

    testcase a3 = { 1, 0 , "1"};
    valid_case(a3);

    testcase a4 = { 1, 0 , "a.a.a.a"};
    valid_case(a4);

    testcase a5 = { 1, 0 , "1.1.1.1.1"};
    valid_case(a5);

    //valid ip
    testcase a6 = { 0, 1, "0.0.0.0"};
    valid_case(a6);
    
    testcase a7 = { 0, 1, "255.0.0.0"};
    valid_case(a7);

    testcase a8 = { 0, 1, "255.255.255.255"};
    valid_case(a8);
}


int main(int argc, char* argv)
{
    test_check_ip();
}
```

Success for case 255.255.255.255
